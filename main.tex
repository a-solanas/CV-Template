% Full-width section heading
\newcommand{\cvsection}[1]{%
  \vspace{10pt}%
  {\color{accent}\small\bfseries\MakeUppercase{#1}}%
  \par\nobreak\vspace{2pt}%
  {\color{accent!50}\hrule height 0.8pt}%
  \vspace{5pt}%
}

% \cvjob{Title}{Period}{Company}{Location}{Promotion}{datelevel}
% datelevel: t = title line, c = company line (default)
\newcommand{\cvjob}[6]{%
  \ifx c#6
    \textbf{#1}\\[1pt]
    {\color{textmuted}\small #3\ifx\relax#4\relax\else{\ ,\ #4}\fi}\hfill{\color{textmuted}\small #2}%
  \else
    \textbf{#1}\hfill{\color{textmuted}\small #2}%
    \ifx\relax#3\relax\else\\[1pt]{\color{textmuted}\small #3\ifx\relax#4\relax\else{\ ,\ #4}\fi}\fi%
  \fi
  \ifx\relax#5\relax\else\\[2pt]{\color{promoColor}\small\itshape #5}\fi%
  \par\vspace{3pt}%
}

% Education entry
% \cvedu{Degree}{Period}{Institution}
\newcommand{\cvedu}[3]{%
  \begin{minipage}[t]{0.56\linewidth}
    \raggedright\small\textbf{#1}
  \end{minipage}%
  \begin{minipage}[t]{0.44\linewidth}
    \raggedleft{\color{textmuted}\small #2}
  \end{minipage}\\[2pt]
  {\color{textmuted}\small #3}%
  \par\vspace{7pt}%
}

% Skill pill
\newcommand{\pill}[1]{%
  \tikz[baseline=(t.base)]{
    \node[
      fill=pillbg,
      draw=pillborder,
      line width=0.5pt,
      rounded corners=3pt,
      inner xsep=5pt,
      inner ysep=2.5pt,
      font=\small,
      text=textmain
    ](t){#1};
  }\hspace{1pt}%
}

% Skill group
\newcommand{\skillgroup}[2]{%
  {\small\bfseries\color{textmain}#1}\\[4pt]
  #2\par\vspace{5pt}%
}

% Publication entry
% \cvpub{Title}{URL}{Venue}{Date}
\newcommand{\cvpub}[4]{%
  \ifx\relax#2\relax
    {\small\textbf{#1}}%
  \else
    {\small\textbf{\href{#2}{#1}}}%
  \fi\\[1pt]
  {\color{textmuted}\small #3\ifx\relax#4\relax\else{\ \textperiodcentered\ #4}\fi}%
  \par\vspace{5pt}%
}

% Course / Certification entry
% \cvcert{Name}{URL}{Entity}{Date}
\newcommand{\cvcert}[4]{%
  \ifx\relax#2\relax
    {\small\textbf{#1}}%
  \else
    {\small\textbf{\href{#2}{#1}}}%
  \fi\\[1pt]
  {\color{textmuted}\small #3\hfill #4}%
  \par\vspace{5pt}%
}
